### 1.3 `script.js`

```js
// ---------- DATA & STATE ----------
const BILLING = {
  MONTHLY: "monthly",
  YEARLY: "yearly",
};

const PLANS = {
  arcade: { label: "Arcade", monthly: 9, yearly: 90 },
  advanced: { label: "Advanced", monthly: 12, yearly: 120 },
  pro: { label: "Pro", monthly: 15, yearly: 150 },
};

const ADDONS = {
  "online-service": {
    label: "Online service",
    description: "Access to multiplayer games",
    monthly: 1,
    yearly: 10,
  },
  "larger-storage": {
    label: "Larger storage",
    description: "Extra 1TB of cloud save",
    monthly: 2,
    yearly: 20,
  },
  "custom-profile": {
    label: "Customizable profile",
    description: "Custom theme on your profile",
    monthly: 2,
    yearly: 20,
  },
};

const formState = {
  currentStep: 1,
  billing: BILLING.MONTHLY,
  personalInfo: {
    fullName: "",
    email: "",
    phone: "",
  },
  planId: "arcade",
  addons: new Set(),
};

// ---------- DOM LOOKUPS ----------
const bodyEl = document.body;
const stepSections = Array.from(document.querySelectorAll(".step"));
const stepIndicators = Array.from(document.querySelectorAll(".step-indicator"));

const backBtn = document.getElementById("back-btn");
const nextBtn = document.getElementById("next-btn");
const confirmBtn = document.getElementById("confirm-btn");

const fullNameInput = document.getElementById("full-name");
const emailInput = document.getElementById("email");
const phoneInput = document.getElementById("phone");

const planInputs = Array.from(document.querySelectorAll(".plan-input"));
const billingToggleBtn = document.getElementById("billing-toggle");
const billingLabels = Array.from(document.querySelectorAll(".billing-label"));

const addonInputs = Array.from(document.querySelectorAll(".addon-input"));

const summaryPlanNameEl = document.getElementById("summary-plan-name");
const summaryPlanPriceEl = document.getElementById("summary-plan-price");
const summaryAddonsEl = document.getElementById("summary-addons");
const summaryTotalLabelEl = document.getElementById("summary-total-label");
const summaryTotalPriceEl = document.getElementById("summary-total-price");
const changePlanBtn = document.getElementById("change-plan-btn");

const themeToggleBtn = document.getElementById("theme-toggle");

// ---------- HELPERS ----------
function formatPrice(value, billing) {
  const suffix = billing === BILLING.MONTHLY ? "/mo" : "/yr";
  return `$${value}${suffix}`;
}

function getPlanPrice(planId, billing) {
  const plan = PLANS[planId];
  return billing === BILLING.MONTHLY ? plan.monthly : plan.yearly;
}

function getAddonPrice(addonId, billing) {
  const addon = ADDONS[addonId];
  return billing === BILLING.MONTHLY ? addon.monthly : addon.yearly;
}

function getErrorElement(id) {
  return document.querySelector(`[data-error-for="${id}"]`);
}

// ---------- VALIDATION ----------
function validateStep(step) {
  let isValid = true;

  // Clear previous errors
  [fullNameInput, emailInput, phoneInput].forEach((input) => {
    input.classList.remove("field-input--error");
  });
  getErrorElement("full-name").textContent = "";
  getErrorElement("email").textContent = "";
  getErrorElement("phone").textContent = "";
  const planError = getErrorElement("plan");
  if (planError) planError.textContent = "";

  if (step === 1) {
    const fullName = fullNameInput.value.trim();
    const email = emailInput.value.trim();
    const phone = phoneInput.value.trim();

    if (!fullName) {
      isValid = false;
      fullNameInput.classList.add("field-input--error");
      getErrorElement("full-name").textContent = "Please enter your name.";
    }

    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!email) {
      isValid = false;
      emailInput.classList.add("field-input--error");
      getErrorElement("email").textContent = "Email is required.";
    } else if (!emailPattern.test(email)) {
      isValid = false;
      emailInput.classList.add("field-input--error");
      getErrorElement("email").textContent = "Please enter a valid email.";
    }

    if (!phone) {
      isValid = false;
      phoneInput.classList.add("field-input--error");
      getErrorElement("phone").textContent = "Please enter your phone number.";
    }

    if (isValid) {
      formState.personalInfo.fullName = fullName;
      formState.personalInfo.email = email;
      formState.personalInfo.phone = phone;
    }
  }

  if (step === 2) {
    const selectedPlanInput = planInputs.find((input) => input.checked);
    if (!selectedPlanInput) {
      isValid = false;
      getErrorElement("plan").textContent = "Please select a plan.";
    } else {
      formState.planId = selectedPlanInput.dataset.planId;
    }
  }

  // Step 3 doesn't require validation (add-ons are optional)
  return isValid;
}

// ---------- STEP NAVIGATION ----------
function goToStep(stepNumber) {
  formState.currentStep = stepNumber;

  stepSections.forEach((section) => {
    const isActive = Number(section.dataset.step) === stepNumber;
    section.classList.toggle("is-active", isActive);
  });

  stepIndicators.forEach((indicator) => {
    const step = Number(indicator.dataset.step);
    indicator.classList.toggle("is-active", step === stepNumber);
  });

  backBtn.disabled = stepNumber === 1 || stepNumber === 5;
  nextBtn.hidden = stepNumber >= 4;
  confirmBtn.hidden = stepNumber !== 4;

  if (stepNumber === 4) {
    updateSummary();
  }
}

function goToNextStep() {
  if (!validateStep(formState.currentStep)) return;
  if (formState.currentStep >= 4) return;
  const target = formState.currentStep + 1;
  goToStep(target);
}

function goToPreviousStep() {
  if (formState.currentStep <= 1) return;
  const target = formState.currentStep - 1;
  goToStep(target);
}

// ---------- SUMMARY ----------
function updateSummary() {
  const billing = formState.billing;
  const plan = PLANS[formState.planId];
  const planPrice = getPlanPrice(formState.planId, billing);

  summaryPlanNameEl.textContent = `${plan.label} (${billing === BILLING.MONTHLY ? "Monthly" : "Yearly"})`;
  summaryPlanPriceEl.textContent = formatPrice(planPrice, billing);

  // Add-ons
  summaryAddonsEl.innerHTML = "";
  let addonsTotal = 0;

  formState.addons.forEach((addonId) => {
    const addon = ADDONS[addonId];
    const price = getAddonPrice(addonId, billing);
    addonsTotal += price;

    const row = document.createElement("div");
    row.className = "summary-addon-row";
    const labelEl = document.createElement("span");
    labelEl.textContent = addon.label;
    const priceEl = document.createElement("span");
    priceEl.className = "summary-addon-row__price";
    priceEl.textContent = `+${formatPrice(price, billing)}`;
    row.append(labelEl, priceEl);
    summaryAddonsEl.appendChild(row);
  });

  const total = planPrice + addonsTotal;
  const intervalText = billing === BILLING.MONTHLY ? "per month" : "per year";
  summaryTotalLabelEl.textContent = `Total (${intervalText})`;
  summaryTotalPriceEl.textContent = formatPrice(total, billing);
}

// ---------- BILLING TOGGLE ----------
function updateBillingVisuals() {
  billingLabels.forEach((label) => {
    const billingType = label.dataset.billing;
    label.classList.toggle("billing-label--active", billingType === formState.billing);
  });

  // Update plan and addon prices in UI
  Object.keys(PLANS).forEach((planId) => {
    const priceEl = document.querySelector(`[data-price-plan="${planId}"]`);
    const tagEl = document.querySelector(`[data-plan-tag="${planId}"]`);
    if (!priceEl || !tagEl) return;

    const price = getPlanPrice(planId, formState.billing);
    priceEl.textContent = formatPrice(price, formState.billing);
    if (formState.billing === BILLING.YEARLY) {
      tagEl.textContent = "2 months free";
    } else {
      tagEl.textContent = "";
    }
  });

  Object.keys(ADDONS).forEach((addonId) => {
    const priceEl = document.querySelector(`[data-price-addon="${addonId}"]`);
    if (!priceEl) return;
    const price = getAddonPrice(addonId, formState.billing);
    priceEl.textContent = `+${formatPrice(price, formState.billing)}`;
  });
}

function toggleBilling() {
  formState.billing =
    formState.billing === BILLING.MONTHLY ? BILLING.YEARLY : BILLING.MONTHLY;
  document.body.setAttribute("data-billing", formState.billing);
  updateBillingVisuals();
  if (formState.currentStep === 4) {
    updateSummary();
  }
}

// ---------- ADD-ONS ----------
function handleAddonChange(event) {
  const checkbox = event.target;
  if (!checkbox.classList.contains("addon-input")) return;

  const addonId = checkbox.dataset.addonId;
  if (!addonId) return;

  if (checkbox.checked) {
    formState.addons.add(addonId);
  } else {
    formState.addons.delete(addonId);
  }
}

// ---------- THEME ----------
function loadThemePreference() {
  const saved = window.localStorage.getItem("msf-theme");
  if (saved === "dark") {
    bodyEl.classList.add("theme-dark");
    bodyEl.classList.remove("theme-light");
  } else {
    bodyEl.classList.add("theme-light");
    bodyEl.classList.remove("theme-dark");
  }
}

function toggleTheme() {
  const isDark = bodyEl.classList.toggle("theme-dark");
  bodyEl.classList.toggle("theme-light", !isDark);
  window.localStorage.setItem("msf-theme", isDark ? "dark" : "light");
}

// ---------- INITIALIZATION ----------
function initForm() {
  loadThemePreference();

  // Initialize billing visuals and plan/addon prices
  document.body.setAttribute("data-billing", formState.billing);
  updateBillingVisuals();

  // Personal info live updates (so going back keeps values)
  fullNameInput.addEventListener("input", (e) => {
    formState.personalInfo.fullName = e.target.value;
  });
  emailInput.addEventListener("input", (e) => {
    formState.personalInfo.email = e.target.value;
  });
  phoneInput.addEventListener("input", (e) => {
    formState.personalInfo.phone = e.target.value;
  });

  // Step navigation
  backBtn.addEventListener("click", () => {
    if (formState.currentStep === 5) {
      // From thank-you screen, don't go back
      return;
    }
    goToPreviousStep();
  });

  nextBtn.addEventListener("click", () => {
    goToNextStep();
  });

  confirmBtn.addEventListener("click", () => {
    if (!validateStep(4)) return;
    goToStep(5);
  });

  // Plan selection
  planInputs.forEach((input) => {
    input.addEventListener("change", () => {
      formState.planId = input.dataset.planId;
      if (formState.currentStep === 4) {
        updateSummary();
      }
    });
  });

  // Billing toggle
  billingToggleBtn.addEventListener("click", toggleBilling);

  // Add-ons
  addonInputs.forEach((input) => {
    input.addEventListener("change", handleAddonChange);
  });

  // Change plan from summary
  changePlanBtn.addEventListener("click", () => {
    goToStep(2);
  });

  // Theme
  themeToggleBtn.addEventListener("click", toggleTheme);

  // Start at step 1
  goToStep(1);
}

document.addEventListener("DOMContentLoaded", initForm);
```

- **Theme toggle with persistence**: Light/dark mode using CSS variables and `localStorage`.
- **Animated step transitions**: Steps fade and slide in when you navigate.
- **Rich layout & micro-interactions**: Gradient background, glass-like card, hover states, and subtle shadows.
- **Dynamic pricing view**: Monthly/yearly toggle updates plan and add-on prices plus summary totals.

### 2.2 HTML (`index.html`) explained

- **`<!DOCTYPE html>` and `<html lang="en">`**:
  - These tell the browser this is a modern HTML5 document and that the language is English.
- **`<meta charset="UTF-8">` and `<meta name="viewport" ...>`**:
  - `charset="UTF-8"` ensures characters display correctly.
  - `viewport` makes the layout responsive on mobile devices.
- **`<link rel="stylesheet" href="styles.css">`**:
  - Connects the HTML to your CSS file so your styles apply.

- **`<body class="theme-light">`**:
  - The `class` defines the starting theme.
  - JavaScript toggles `theme-light` / `theme-dark` to switch colors.

- **Header (`<header class="app-header">`)**:
  - Contains:
    - A logo text (`multiStepWizard`) – this is purely visual branding.
    - A **theme toggle button** (`id="theme-toggle"`) – JavaScript attaches a click handler to this to switch themes.

- **Main container (`<main class="app">`)**:
  - Wraps the entire interactive card.
  - Inside is `.app-card`, split into:
    - `aside.steps-sidebar` – shows the step numbers and labels.
    - `section.form-area` – contains the actual `<form>`.

- **Sidebar (`<aside class="steps-sidebar">`)**:
  - Uses an ordered list `<ol class="steps-list">`.
  - Each `<li class="step-indicator" data-step="1">`:
    - `data-step="1"` is a **custom attribute** (“data attribute”) used by JavaScript to link indicators to step sections.
    - The `is-active` class marks the current step; CSS styles it differently.

- **Form (`<form id="multi-step-form" novalidate>`)**:
  - `id` lets JavaScript find it.
  - `novalidate` tells the browser not to show its own built-in validation bubbles; we handle errors manually.

- **Step sections (`<section class="step" data-step="1">`)**:
  - Each step is a `<section>` with:
    - `class="step"` – gives common styling.
    - `data-step="1"` – matches the sidebar’s step and lets JavaScript show/hide the correct section.
    - A heading (`<h1>`) and description `<p>` to explain what the user should do.
  - The first step also has `class="step is-active"` so it’s visible on load.

- **Form fields**:
  - Each field is in a `.field-group`:
    - `<label>` connects text to its `<input>` using `for="full-name"` and `id="full-name"`.
    - `.field-input` gives consistent styling.
    - The `required` attribute marks the field as mandatory.
    - A `<p class="field-error" data-error-for="full-name"></p>` is a placeholder for error messages that JavaScript can fill.

- **Plan selection (Step 2)**:
  - Each plan is a `<label class="plan-card">` that wraps:
    - A hidden radio `<input type="radio" name="plan" data-plan-id="arcade">`.

  - Clicking the card checks the radio button automatically because the `<input>` sits inside the `<label>`.
  - `data-plan-id="arcade"` identifies the plan (“arcade”, “advanced”, “pro”) for code logic.
  - Additional elements like `data-price-plan` and `data-plan-tag` are filled by JavaScript with prices and optional tags (e.g., “2 months free”).

  - A wrapper `.billing-toggle` with:
    - Two labels: `<span class="billing-label" data-billing="monthly">` and similar for yearly.
    - A button `id="billing-toggle"` in the center styled like a switch.

### 2.4 JavaScript (`script.js`) explained

#### 2.4.1 Data and state

- **`BILLING` object**:
  - Holds constant strings `"monthly"` and `"yearly"` so you don’t mistype them.

- **`PLANS` object**:
  - Maps plan IDs like `"arcade"` to their display label and monthly/yearly prices.
  - Example: `PLANS.arcade.monthly` is 9; `PLANS.arcade.yearly` is 90.

- **`ADDONS` object**:
  - Similar concept for add-ons: each add-on has a label, description, and prices.

- **`formState` object**:
  - Keeps track of:
    - `currentStep` – which step (1–5) is active.
    - `billing` – `"monthly"` or `"yearly"`.
    - `personalInfo` – an object containing `fullName`, `email`, `phone`.
    - `planId` – which main plan is selected.
    - `addons` – a `Set` (collection) of selected add-on IDs (e.g., `"online-service"`).

Using a central `formState` makes it easier to:

- Update the UI based on one source of truth.
- Build the summary by reading from this object only.

#### 2.4.2 DOM lookups

- `document.querySelector(...)` and `document.getElementById(...)`:
  - These functions select elements from the page:
    - `.querySelector(".step")` finds the first element with class `step`.
    - `.querySelectorAll(".step")` returns all step sections.
    - `.getElementById("back-btn")` finds the single element with that ID.

- We store these elements in variables (e.g., `backBtn`, `stepSections`) so we can reuse them:
  - **Example**: `backBtn.addEventListener("click", ...)` attaches a function to run each time the button is clicked.

#### 2.4.3 Helper functions

- `formatPrice(value, billing)`:
  - Takes a number (`value`, like 9) and a billing type.
  - Returns strings like `"$9/mo"` or `"$90/yr"`.

- `getPlanPrice(planId, billing)` and `getAddonPrice(addonId, billing)`:
  - Look up the correct price from the `PLANS` or `ADDONS` config based on billing mode.

- `getErrorElement(id)`:
  - Uses the `data-error-for` attribute to find the corresponding error `<p>` element for a field.

#### 2.4.4 Validation (`validateStep`)

- `validateStep(step)`:
  - Checks the data in the current step and returns `true` or `false`.
  - For step 1:
    - Reads `fullName`, `email`, `phone` and trims whitespace.
    - If a field is empty, sets `isValid = false`, adds `field-input--error` to the input, and sets the error message text.
    - Uses a simple **regular expression** (`emailPattern`) to detect basic invalid email formats.
    - If all good, saves the values into `formState.personalInfo`.
  - For step 2:
    - Checks that at least one plan radio is checked.
    - If not, shows an error under the plan area.
    - If yes, updates `formState.planId`.
  - Step 3 has no required fields (add-ons are optional).

This function is called **before moving to the next step**. If validation fails, the step doesn’t change.

#### 2.4.5 Step navigation (`goToStep`, `goToNextStep`, `goToPreviousStep`)

- `goToStep(stepNumber)`:
  - Updates `formState.currentStep`.
  - Loops through all `.step` sections and toggles the `is-active` class based on `data-step`.
  - Loops through `.step-indicator` to highlight the active step in the sidebar.
  - Adjusts navigation buttons:
    - Disables “Go back” on step 1 and 5.
    - Hides “Next step” on step 4 (we show “Confirm” instead).
    - Calls `updateSummary()` when entering step 4 so the totals are fresh.

- `goToNextStep()`:
  - Runs `validateStep(formState.currentStep)`.
  - If invalid, stops.
  - If valid and not already on the last step, moves to the next one.

- `goToPreviousStep()`:
  - Moves back a step (unless already on step 1).

These functions give a **clear, centralized way** to control step logic.

#### 2.4.6 Summary (`updateSummary`)

- Reads:
  - `formState.billing` (monthly/yearly).
  - `formState.planId` and `formState.addons`.
- Calculates:
  - Plan price using `getPlanPrice`.
  - Add-ons total by looping over `formState.addons` and calling `getAddonPrice`.
  - Final total = plan price + sum of add-on prices.
- Updates DOM:
  - `summary-plan-name` – writes something like `"Arcade (Monthly)"`.
  - `summary-plan-price` – uses `formatPrice`.
  - `summary-addons` – cleared and rebuilt using DOM methods:
    - `document.createElement("div")`, setting `.className`.
    - `row.append(labelEl, priceEl)`.
    - `summaryAddonsEl.appendChild(row)`.
  - `summary-total-label` – shows `Total (per month)` or `Total (per year)`.
  - `summary-total-price` – final formatted price.

#### 2.4.7 Billing toggle (`updateBillingVisuals`, `toggleBilling`)

- `updateBillingVisuals()`:
  - Highlights the active billing label (monthly/yearly) by toggling `billing-label--active`.
  - Updates all plan and add-on price elements in the UI based on the current billing mode.

- `toggleBilling()`:
  - Flips `formState.billing` between `"monthly"` and `"yearly"`.
  - Updates the `data-billing` attribute on `<body>` for CSS to animate the switch thumb.
  - Calls `updateBillingVisuals()` and `updateSummary()` if currently on the summary step.

This is a **creative enhancement**, letting users see how prices change instantly.

#### 2.4.8 Add-ons (`handleAddonChange`)

- Attached to each `.addon-input` checkbox.
- When a checkbox changes:
  - Gets the `data-addon-id`.
  - If `.checked` is true, adds that ID to the `formState.addons` Set.
  - Otherwise removes it.

Using a `Set` makes it easy to add/remove without worrying about duplicates.

#### 2.4.9 Theme switching (`loadThemePreference`, `toggleTheme`)

- `loadThemePreference()`:
  - Reads `msf-theme` from `localStorage`.
  - Applies either `.theme-dark` or `.theme-light` to the `body`.
  - This means the theme setting is remembered when you refresh the page.

- `toggleTheme()`:
  - Toggles `.theme-dark` on `body` and the opposite `.theme-light`.
  - Stores the new preference in `localStorage`.

This is another **creative improvement** making the app feel modern.

#### 2.4.10 Initialization (`initForm` and `DOMContentLoaded`)

- `initForm()`:
  - Runs once, when the page is fully loaded.
  - Steps:
    1. Calls `loadThemePreference()` and `updateBillingVisuals()`.
    2. Attaches `input` listeners to text fields so changes are reflected in `formState.personalInfo`.
    3. Sets up click handlers for:
       - Back, Next, and Confirm buttons.
       - Plan radio inputs.
       - Billing toggle button.
       - Add-on checkboxes.
       - “Change” button in the summary (jumps back to step 2).
       - Theme toggle button.
    4. Calls `goToStep(1)` to show the first step.

- `document.addEventListener("DOMContentLoaded", initForm);`:
  - Waits until the HTML is fully loaded before running `initForm`.
  - Ensures that all the `document.getElementById` and `querySelector` calls can actually find the elements they need.

### 1.1 HTML – `index.html`

**Purpose**: Holds the full multi-step form, progress sidebar, and final confirmation.

- **Page shell**
  - A single HTML file, `index.html`, that contains:

  - An `<aside>` that shows:

  - You only show **one step at a time** using CSS classes controlled by JavaScript.
  - **Creative upgrade**: Wrap each step in a `<section>` with `aria-labelledby` pointing to that step’s heading, improving accessibility.

- **Navigation buttons**
  - At the bottom of the form:
    - A “Go back” button (hidden on the first step).
    - A “Next step” button.
    - A “Confirm” button on the last step.
  - **Creative upgrade**: Use `type="button"` for navigation buttons and leave `type="submit"` only for the final confirm, so behavior is explicit and easy to control.

- **Final thank-you state**
  - A final `<section>` inside the same form container that replaces the summary once the user clicks “Confirm”.
  - **Creative upgrade**: Show a **subtle success illustration** (icon) and a short recap of their choices.

### 1.2 CSS – `base.css`, `layout.css`, `components.css`, `themes.css`

#### `base.css`

**Purpose**: Global resets and design tokens.

- Include:
  - A modern CSS reset (box-sizing, margin: 0, etc.).
  - Custom properties (CSS variables) for:
    - Colors (primary, accent, danger, background, text).
    - Spacing scale (e.g., `--space-1`, `--space-2`…).
    - Font sizes and font families.
  - **Creative upgrade**: Define a **design system** via CSS variables so you can change colors, spacing, or fonts in one place.

#### `layout.css`

**Purpose**: Page layout and responsive behavior.

- Use:
  - A **mobile-first** single-column layout.
  - A **CSS Grid** or Flexbox layout that switches to two columns (sidebar + content) on larger screens.
  - Utility classes such as `.container`, `.grid-2`, `.stack`, `.inline-center`.
  - **Creative upgrade**: Add a **max-width container** with centered content and generous padding to make the UI look premium on big screens.

#### `components.css`

**Purpose**: Visual styles for UI components.

- Buttons:
  - Primary button style for “Next” / “Confirm”.
  - Ghost or outline style for “Go back”.
  - Disabled state and focus styles.
  - **Creative upgrade**: Add **micro-interactions** (smooth hover transitions, pressed states) for more tactile feel.

- Inputs:
  - Consistent padding, border radius, and font.
  - Error state class for invalid fields (e.g., red border).
  - Helper text and error messages styled clearly below inputs.
  - **Creative upgrade**: Show error messages with a **subtle fade-in animation** instead of just appearing abruptly.

- Step cards and summary:
  - Card-like appearance for each plan / add-on.
  - Summary card with sections for plan, add-ons, total.
  - **Creative upgrade**: Use **soft shadows and layered cards** to visually separate different information blocks.

#### `themes.css`

**Purpose**: Light/dark theme support and custom accents.

- Light theme:
  - Light backgrounds, strong contrast text.
- Dark theme:
  - Dark backgrounds, soft text, and accent highlights.
- Theme toggle:
  - A small toggle in the header or footer that switches themes by toggling a class on `<body>`.
  - **Creative upgrade**: Implement a **smooth color transition** when switching themes (`transition: background-color, color`).

### 1.3 JavaScript – `state.js`, `validation.js`, `steps.js`, `summary.js`, `storage.js`, `main.js`

Keep your JavaScript in **small, focused modules** so it’s easier to understand and maintain.

#### `state.js`

**Purpose**: Holds the current form data and active step.

- Responsibilities:
  - Keep an object for user data (name, email, phone, plan, add-ons, billing type, etc.).
  - Track which step is active.
  - Provide simple functions like “go to next step”, “go to previous step”, “update a field”.
  - **Creative upgrade**: Make state updates **pure and predictable** (functions that always return a new state based on the old one) so debugging is easier.

#### `validation.js`

**Purpose**: Validates user input.

- Responsibilities:
  - Validate text fields (required, minimum length).
  - Validate email format.
  - Ensure required selections are made (plan, etc.).
  - Provide functions that accept data and return:
    - A boolean (valid / invalid).
    - A list of error messages keyed by field.
  - **Creative upgrade**: Keep all validation rules in **one place** so you can easily adjust them later without digging through multiple files.

#### `steps.js`

**Purpose**: Controls which step is visible and updates the UI when the step changes.

- Responsibilities:
  - Show/hide step sections.
  - Update the active state in the sidebar (highlight current step).
  - Enable/disable buttons based on the current step.
  - **Creative upgrade**: Add **animated transitions** between steps (slide, fade, or scale) for a smoother experience.

#### `summary.js`

**Purpose**: Builds and updates the summary view before confirmation.

- Responsibilities:
  - Read current state and render:
    - Selected plan and billing period.
    - Selected add-ons with prices.
    - Total price calculation.
  - React when billing period or plan selection changes.
  - **Creative upgrade**: Allow users to **edit from the summary** by clicking a “Change” link that jumps them back to the relevant step and brings them forward again automatically when they’re done.

#### `storage.js`

**Purpose**: Optional persistence layer using `localStorage`.

- Responsibilities:
  - Save the current state to `localStorage` on each step.
  - Load saved state when the app starts.
  - Provide a function to reset stored data on successful confirmation.
  - **Creative upgrade**: Implement **autosave** so that if the user refreshes the page or comes back later, their progress is restored.

#### `main.js`

**Purpose**: Entry point that wires everything together.

- Responsibilities:
  - On page load:
    - Initialize state (possibly from `storage.js`).
    - Attach event listeners to buttons, form fields, and theme toggle.
    - Render the initial step and apply the correct theme.
  - Delegate logic to other modules (`steps.js`, `validation.js`, `summary.js`, `storage.js`).
  - **Creative upgrade**: Centralize all event binding here, so every interaction is easy to discover and debug.

  - Add a simple toggle control (for example, a checkbox or switch) in the header.
- In `main.js` (you will wire this up later):
  - Plan to read the toggle state and add or remove the `.theme-dark` class on `<body>`.
  - Plan to store the user’s choice using `localStorage` so the theme persists across reloads.

### Step 6 – Plan your JavaScript behavior

Before writing any code, outline what needs to happen:

- When the page loads:
  - Load saved data (if any) from storage.
  - Set the starting step to 1.
  - Render the corresponding step section and update the sidebar.
  - Apply the correct theme based on saved preference.
- When the user fills in a step and clicks “Next step”:
  - Validate that step’s fields.
  - Show any error messages under the relevant inputs.
  - If valid, save the data to your state object.
  - Move to the next step and update the UI.
- When the user clicks “Go back”:
  - Move backwards one step and show the previous fields with already-filled values.
- When the user reaches the summary:
  - Build the summary content from the state, including prices and total.
  - Allow them to change their plan or add-ons by jumping back.
- When the user clicks “Confirm”:
  - Optionally perform a final validation.
  - Show the final thank-you section and clear stored data.

### Step 7 – Implement state management (`state.js`)

- Create a simple object to hold:
  - Current step number.
  - Personal info fields.
  - Plan selection (name, price, billing period).
  - Add-ons (which ones are selected and their prices).
- Write small functions that:
  - Get and update values inside this state.
  - Change the current step (for example, increment or decrement the step).
- Keep these functions **independent of the DOM** so they are easy to test mentally: they just receive data and return updated data.

### Step 8 – Implement validation (`validation.js`)

- For each step, write a function that:
  - Receives the part of the state that belongs to that step.
  - Checks each required field and returns:
    - An indication of whether it’s valid.
    - A set of error messages for invalid fields.
- For example:
  - Step 1: require non-empty name, valid email format, and a phone number.
  - Step 2: ensure a plan is selected.
  - Step 3: allow zero or more add-ons (but you might encourage at least one).
- Decide how you will show errors:
  - For each field, you can reserve a small `<span>` under it and fill it with the error text when needed.

### Step 9 – Control visible steps (`steps.js`)

- Write functions that:
  - Show the correct step section and hide the others (for example, by adding or removing a CSS class like `is-hidden`).
  - Update the sidebar:
    - Mark the current step as active.
    - Optionally mark previous steps as “completed”.
  - Enable or disable the “Go back” and “Next step” buttons depending on the step.
- If you want animated transitions:
  - Add or remove CSS classes that trigger transitions (for example, fade-in or slide-in).

### Step 10 – Build the summary logic (`summary.js`)

- Decide how to represent prices and totals:
  - Store plan and add-on prices in your state or a separate configuration.
  - When you build the summary, calculate the total by combining the selected plan and all selected add-ons.
- Write a function that:
  - Reads the current state.
  - Constructs summary items (plan, add-ons, total).
  - Inserts them into the summary container in the DOM.
- Add a “Change” link near the plan section:
  - When clicked, it should send the user back to the plan step.

### Step 11 – Add storage and autosave (`storage.js`)

- Decide what to store:
  - All user data and the current step.
  - The selected theme.
- Write functions to:
  - Save the current state to `localStorage` every time the user completes a step or changes important data.
  - Load and parse saved data on page load.
  - Clear storage when the user finishes the process.
- This gives your app a **professional feel**: users don’t lose their progress if they refresh the page.

### Step 12 – Wire everything together (`main.js`)

- In `main.js`, write your initialization logic:
  - Load data from `storage.js`.
  - Initialize the state using `state.js`.
  - Set the active step and call functions from `steps.js` to show the correct UI.
  - Attach event listeners to:
    - Form fields (update state on input or change).
    - Navigation buttons (validate, save, and move steps).
    - Summary “Change” links (jump back to specific steps).
    - Theme toggle (switch themes and save preference).
- Each event handler should:
  - Use `validation.js` to check data where needed.
  - Use `state.js` to update internal data.
  - Use `steps.js` and `summary.js` to update what the user sees.
  - Use `storage.js` to persist progress.

### Step 13 – Polish, test, and refine

- Test on different screen sizes:
  - Narrow mobile width.
  - Tablet size.
  - Desktop width.
- Try navigating only with the keyboard:
  - Make sure focus states are visible on buttons and inputs.
  - Check that pressing Enter behaves as expected.
- Intentionally make mistakes:
  - Leave fields empty and go to the next step.
  - Choose invalid email formats.
  - Confirm that clear error messages appear without breaking the layout.
- Review your creative upgrades:
  - Light/dark themes switch smoothly.
  - Step transitions feel responsive but not distracting.
  - Autosave works and restores your progress.

## 9. Example Interactions

### Example 1: Architecture discussion

**User:** I'm debating whether to use CSS-in-JS or vanilla CSS with custom properties for the component styling. What do you think?

**Mentor response:**
"Depends on what you're optimizing for.

**CSS-in-JS trade-offs:**
- Co-location of styles and components (great DX, easy to reason about)
- Runtime cost (varies by library, but it exists)
- TypeScript integration for props-based styling
- Bundle implications (critical CSS extraction helps, but adds complexity)

**Custom properties + vanilla CSS:**
- Zero runtime cost
- Leverages the cascade (feature or bug depending on your perspective)
- Requires more discipline around organization
- Theming is straightforward with properties

In a portfolio context, showing you can architect vanilla CSS well actually demonstrates more than reaching for a library. That said, if you're targeting companies that use CSS-in-JS, demonstrating familiarity is valid.

What's the priority here - showcasing CSS fundamentals, or demonstrating you can work in a specific toolchain?"

### Example 2: Performance edge case

**User:** I'm animating a list of items on scroll. It works but feels a bit janky on lower-end devices.

**Mentor response:**
"Scroll-linked animations are performance-sensitive. A few angles to probe:

**What's triggering the jank:**
- Are you animating properties that trigger layout (width, height, top, left)?
- Are you doing work on every scroll event, or throttling/debouncing?
- How many elements are you animating simultaneously?

**Things to check:**
- Use DevTools Performance panel - look for long frames and layout thrashing
- `will-change` can help, but use it sparingly (it's not free)
- `transform` and `opacity` are the cheapest properties to animate
- Consider `Intersection Observer` instead of scroll events if you're triggering animations at thresholds

**Modern alternative:**
- `scroll-timeline` in CSS is getting good support and moves this work off the main thread entirely

Have you profiled it yet? The Performance panel will tell you exactly where the time is going."

### Example 3: Challenging an approach

**User:** I'm going to store all the app state in localStorage and sync it on every change for persistence.

**Mentor response:**
"That'll work, but let me push back a bit.

**Concerns with sync-on-every-change:**
- `localStorage` is synchronous and blocks the main thread
- Frequent writes can add up, especially on slower devices
- Storage events can fire across tabs, which might cause unexpected behavior

**Things to consider:**
- Debounce writes so you're not hitting storage on every keystroke
- Consider what actually needs to persist vs. what's session-specific
- Handle storage limits and quota exceeded errors
- What happens if localStorage is disabled or full?

**Alternative angle:**
- If this is complex state, IndexedDB is async and more robust (though more verbose)
- Or persist on specific events (blur, page unload) rather than every change

What's the actual persistence requirement? That drives whether this approach is appropriate or overkill."

### Example 4: Quick validation

**User:** I'm using `aria-live="polite"` for the form validation messages. Sound right?

**Mentor response:**
"That's correct for non-urgent feedback. `polite` waits for a pause in screen reader speech before announcing.

One nuance: if you're showing multiple validation errors at once, consider whether they should all be in a single live region (announced together) or if there's a summary approach that's less verbose.

Also, make sure the live region exists in the DOM before the content changes - dynamically inserted live regions can be ignored by some screen readers.

You're on the right track."
